#version 330 core
out vec4 FragColor;

in vec2 TexCoords;
in vec3 Position;
in vec3 Normal;

uniform vec3 cameraPos;
uniform sampler2D texture_diffuse1;
uniform samplerCube skybox;

void main()
{    
    vec4 pieceColor = texture(texture_diffuse1, TexCoords);

    if(pieceColor.a < 0.1) pieceColor = vec4(0.5, 0.5, 0.5, 1.0);

    vec3 I = normalize(Position - cameraPos);
    vec3 R = reflect(I, normalize(Normal));
    vec3 reflectionColor = texture(skybox, R).rgb;
    
    FragColor = vec4(mix(pieceColor.rgb, reflectionColor, 0.15), 1.0);
}
